Add an App.config file with the following app settings

<?xml version="1.0" encoding="utf-8" ?>
<configuration>
  <appSettings>
    <add key="AppKey" value="" />
    <add key="CertificateLocation" value="Betfair.p12" />
    <add key="CertificatePassword" value="" />
    <add key="Username" value="" />
    <add key="Password" value="" />
  </appSettings>
</configuration>
